<!DOCTYPE html>

<!--     要加 enctype 为 multipart/form-data 否则无法提交文件！-->

<!--    搜索框 search-->
    百度一下<input type="search"> <input type="submit">
</form>
<hr/>
<!-- 用get 提交 地址栏上会有信息 用post 则会隐藏此信息-->
<!-- 而 action 里的地址则会在提交后跳转到此页面-->
<form method="post" action="success.html">
<!--    账号 text | 名字为 user | 默认值为 张三 | 尺寸size 为 20 | 提交长度上限 10  -->
    <p>账号:<input name="user"  type="text" size="20" maxlength="10" placeholder="请输入账号" /></p>
<!--    密码 password-->
    <p>密码:<input name="password" value="123456" type="password" size="20" maxlength="10"/></p>
<!--    年龄-->
    <p>年龄:<input type="number"/></p>
<!--    最大值 最小值  |   类型为数字  | 最大值为 100 | 最小值为 0 | 默认值为 20-->
    <p>最大值最小值测试:<input type="number" max="100" min="0" value="20"></p>
<!--    emil-->
    <p>邮箱:<input type="email"/> </p>
<!--    隐藏域  用于 程序员测试-->
    <input type="hidden" value="123">
<!--    文本域-->
    <p>简介：<textarea cols="20" rows="5"></textarea> </p>
<!--    身高-->
    <p>身高:<input type="range"/></p>

        <input type="checkbox" name="hobby" />篮球

    <p><input type="button"   value="普通按钮"/></p>

    <p><input type="button"   disabled value="普通按钮"/></p>

    <p><input type="button"   readonly value="普通按钮"/></p>

CSS

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>CSS</title>
<!--    内部样式-->
    <style>
        h1{
            color: red;
            font-size: 40px;
        }
    </style>
<!--    外部样式表的引入 链接式-->
<!--    使用外部样式表 | 链接 |非必填-->
    <link rel="stylesheet" href="Text01_css.css" type="text/css"/>
<!--    外部样式表的引入 导入式-->
<!--    @导入 | url（地址）-->
    <style>
        @import "经典练习/Text01_css.css";
    /*    @import url(Text01_css.css)*/
    /*    完整规范写法*/
    </style>
</head>
<body>
<!-- 加多个样式用 ; 隔开-->
<!--行内样式 内联样式-->
<p style="color: green; font-size: 30px;">教育改变生活</p>
<hr/>
<!--内部样式表 只要是h标签都会被改变相同的样式-->
<!-- 就近原则 当内联有 那么外部样式被覆盖-->
<h1 style="color: green">教育改变生活</h1>
<h1>教育改变生活</h1>
<h1>教育改变生活</h1>
<h1>教育改变生活</h1>
